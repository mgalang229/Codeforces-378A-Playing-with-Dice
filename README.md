  d = 0

    if(2 - 1 < 5 - 1) // true

    if(2 - 3 < 5 - 3) // true

    else if(5 - 4 < 2 - 4) // true
      s -> 1
      
  * Fifth Loop *
    if(2 - 5 < 5 - 5) // false
    else if(5 - 5 < 2 - 5) // true
      s -> 2

    else if(5 - 6 < 2 - 6) // true
